\def\SetCatCode#1{%
 \loop\ifnum\ChrA<\ChrB
  \catcode\ChrA=#1
  \sfcode\ChrA=1000
  \lccode\ChrA=\ChrA
  \advance\ChrA 1\relax \repeat
}
